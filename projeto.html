# FUN√á√ÉO PARA CONTAR DEDOS

def contar_dedos(landmarks_mao):
    """
    Conta quantos dedos est√£o levantados
    Retorna: n√∫mero de dedos (0 a 5)
    """
    dedos = 0
    
    # √çndices das pontas dos dedos no MediaPipe
    # Polegar: 4, Indicador: 8, M√©dio: 12, Anelar: 16, M√≠nimo: 20
    pontas = [4, 8, 12, 16, 20]
    
    # POLEGAR - compara√ß√£o no eixo X
    # Para m√£o direita: ponta do polegar mais √† esquerda que a base = levantado
    try:
        if landmarks_mao.landmark[pontas[0]].x < landmarks_mao.landmark[pontas[0] - 1].x:
            dedos += 1
    except:
        pass  # Se der erro, ignora
    
    # OUTROS DEDOS - compara√ß√£o no eixo Y
    # Ponta do dedo mais acima que a junta = levantado
    for i in range(1, 5):
        try:
            if landmarks_mao.landmark[pontas[i]].y < landmarks_mao.landmark[pontas[i] - 2].y:
                dedos += 1
        except:
            pass  # Se der erro, ignora
    
    return dedos

/*
 * CONTROLE DE LEDS E SERVO POR DETEC√á√ÉO DE M√ÉOS
 * Arduino Uno
 * 
 * Conex√µes:
 * - Servo motor no pino 9
 * - LED 1 no pino 3 (com resistor 220Œ©)
 * - LED 2 no pino 4 (com resistor 220Œ©)
 * - LED 3 no pino 5 (com resistor 220Œ©)
 * - LED 4 no pino 6 (com resistor 220Œ©)
 * - LED 5 no pino 7 (com resistor 220Œ©)
 * - GND dos LEDs e servo no GND do Arduino
 * - VCC do servo no 5V do Arduino
 */

#include 

// ============================================
// DECLARA√á√ÉO DE VARI√ÅVEIS
// ============================================
Servo myservo;           // Objeto do servo motor
int pos = 0;              // Posi√ß√£o atual do servo
int valorRecebido = 0;    // Valor recebido pela serial
int numeroDedos = 0;      // N√∫mero de dedos detectados

// Pinos dos LEDs (conecte nos pinos 3,4,5,6,7)
const int ledPins[] = {3, 4, 5, 6, 7};
const int numLeds = 5;    // Quantidade de LEDs

// ============================================
// CONFIGURA√á√ÉO INICIAL
// ============================================
void setup() {
  // Inicia comunica√ß√£o serial (mesma velocidade do Python)
  Serial.begin(9600);
  
  // Configura o servo no pino 9
  myservo.attach(9);
  myservo.write(0); // Posi√ß√£o inicial (0 graus)
  
  // Configura os pinos dos LEDs como sa√≠da
  for (int i = 0; i < numLeds; i++) {
    pinMode(ledPins[i], OUTPUT);
    digitalWrite(ledPins[i], LOW); // Todos apagados no in√≠cio
  }
  
  // Mensagem de inicializa√ß√£o (opcional)
  Serial.println("Arduino pronto!");
}

// ============================================
// LOOP PRINCIPAL
// ============================================
void loop() {
  // Verifica se tem dados chegando pela serial
  // O Python envia n√∫meros de 100 a 105 (100 = 0 dedos, 105 = 5 dedos)
  if (Serial.available() >= 3) {
    
    // Buffer para receber os dados (3 d√≠gitos + caractere nulo)
    char dados[4];
    
    // L√™ os dados at√© encontrar uma quebra de linha
    int bytesLidos = Serial.readBytesUntil('\n', dados, 3);
    dados[bytesLidos] = '\0'; // Adiciona caractere nulo no final
    
    // Converte a string para n√∫mero inteiro
    valorRecebido = atoi(dados);
    
    // Verifica se est√° no intervalo v√°lido (100 a 105)
    if (valorRecebido >= 100 && valorRecebido <= 105) {
      
      // Calcula quantos dedos (subtrai 100)
      // 100 = 0 dedos, 101 = 1 dedo, 102 = 2 dedos, etc.
      numeroDedos = valorRecebido - 100;
      
      // ========================================
      // CONTROLE DO SERVO MOTOR
      // ========================================
      // Mapeia o n√∫mero de dedos (0-5) para posi√ß√£o do servo (0-170 graus)
      pos = map(numeroDedos, 0, 5, 0, 170);
      myservo.write(pos); // Move o servo para a posi√ß√£o calculada
      
      // ========================================
      // CONTROLE DOS LEDS
      // ========================================
      // Acende os LEDs de acordo com o n√∫mero de dedos
      // Ex: 3 dedos = acende LEDs 1, 2 e 3
      for (int i = 0; i < numLeds; i++) {
        if (i < numeroDedos) {
          digitalWrite(ledPins[i], HIGH); // Acende LED
        } else {
          digitalWrite(ledPins[i], LOW);  // Apaga LED
        }
      }
      
      // ========================================
      // FEEDBACK NO MONITOR SERIAL
      // ========================================
      Serial.print("Dedos: ");
      Serial.print(numeroDedos);
      Serial.print(" | Servo: ");
      Serial.print(pos);
      Serial.println(" graus");
      
    } else {
      // Se receber um valor inv√°lido, desliga tudo
      for (int i = 0; i < numLeds; i++) {
        digitalWrite(ledPins[i], LOW);
      }
      myservo.write(0);
      Serial.println("Comando invalido!");
    }
  }
  
  delay(10); // Pequena pausa para estabilidade
}
